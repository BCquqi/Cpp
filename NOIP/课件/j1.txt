#include <bits/stdc++.h>
using namespace std;
vector <string> s;  // 定义一个字符串类型的向量 
int main(){
    int n;  // 定义一个整数 n   
    cin >>n;  // 从标准输入读取 n 的值    
    s.resize(n);  // 调整向量 s 的大小为 n    
    for(int i=0;i<n;i++)  // 从 0 到 n-1 进行循环      
      cin >>s[i];  // 从标准输入读取向量 s 中每个元素的值    
    sort(s.begin(),s.end());  // 对向量 s 中的元素进行排序   
    s.erase(unique(s.begin(),s.end()),s.end());  // 去除向量 s 中的重复元素   
    cout <<52-s.size();  // 输出 52 减去向量 s 的大小的值   
 return 0;}

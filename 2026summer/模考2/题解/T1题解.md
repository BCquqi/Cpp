$n$ 和 $m$ 都比较小，可以直接模拟检查。

参考代码：

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int n,m;
string s[52];
bool check(int x,int y,int l)
{
    for(int i=x-l;i<=x+l;i++)  //正对角线
    {
        int j=i-x+y;
        if(i<1||i>n||j<1||j>m)return false;
        if(i==x)continue;
        if(s[i][j] != '\\')return false;
    }
    for(int i=x-l;i<=x+l;i++)
    {
        int j=x+y-i;
        if(i<1||i>n||j<1||j>m)return false;
        if(i==x)continue;
        if(s[i][j] != '/')return false;
    }
    for(int i=x-l;i<=x+l;i++)
    {
        if(i<1||i>n)return false;
        if(i==x)continue;
        if(s[i][y]!='|')return false;
    }
    for(int j=y-l;j<=y+l;j++)
    {
        if(j<1||j>m)return false;
        if(j==y)continue;
        if(s[x][j]!='-')return false;
    }
    return true;    
}
int main()
{
    ios::sync_with_stdio(false); cin.tie(0);
    cin>>n>>m;
    for(int i=1;i<=n;i++)
    {
        cin>>s[i];
        s[i]='0'+s[i];
    }
    int ans=0;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            if(s[i][j]=='+')
            {
                int l=1;
                while(check(i,j,l))l++;
                ans=max(ans,l-1);
            }
    cout<<ans;
    return 0;
}
```

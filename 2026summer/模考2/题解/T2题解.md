#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
string hang[105],lie[105];
int n,m;   
int a,b,c,d;
int cal(int x)
{
   // cout<<x<<" ";
    int ret=0;
    while(x)
    {
        ret++;
        x-=x&(-x);
    }
    //cout<<ret<<"\n";
    return ret;
}
int find(int x,int y,int xx,int yy)
{
    char ch='*';
    if(x==xx)
    {
        if(y>yy)swap(y,yy);
        ch=hang[x][y];
    }
    if(y==yy)
    {
        if(x>xx)swap(x,xx);
        ch=lie[x][y];
    }
    
    //cout<<x<<" "<<y<<" -> "<<xx<<" "<<yy<<" : "<<ch<<"\n";

    if(ch=='P')return 0;
    if(ch=='C')return 1;
    if(ch=='Z')return 2;
    if(ch=='N')return 3;
}
bool check(int col)
{
    queue<pair<int,int> >que;
    bool v[101][101]={0};
    int dx[4]={0,0,1,-1};
    int dy[4]={1,-1,0,0};
    v[a][b]=true; que.push({a,b});
    if(c==a && d==b)return true;
    while(que.size())
    {
        int x=que.front().first,y=que.front().second;   
        que.pop();
        for(int i=0;i<4;i++)
        {
            int xx=x+dx[i],yy=y+dy[i];
            if(xx<1||xx>n||yy<1||yy>m)continue;
            if(v[xx][yy])continue;
            int p=find(x,y,xx,yy);
            //cout<<x<<" "<<y<<" "<<xx<<" "<<yy<<" : "<<c<<"\n\n";
            if((col>>p)&1)
            {
                v[xx][yy]=true;
                que.push({xx,yy});
                if(xx==c && yy==d)return true;
            }
        }
    }
    return false;
}

int main()
{
    ios::sync_with_stdio(false); cin.tie(0);
    cin>>n>>m;
    for(int i=1;i<=n;i++)cin>>hang[i],hang[i]='0'+hang[i];
    for(int i=1;i<n;i++)cin>>lie[i],lie[i]='0'+lie[i];
    int q;
    cin>>q;
    while(q--)
    {
        cin>>a>>b>>c>>d;
        int ans=5;
        for(int i=1;i<(1<<4);i++)
            if(check(i))
            {
                //cout<<"Yes: "<<i<<"\n";
                ans=min(ans,cal(i));
            }
        cout<<ans<<"\n";
    }
    
    return 0;
}

01背包:
有n个体积为w[i],价值为v[i]的物品,问在不超过容量限制下,能取得的物品最大价值.
dp[i][j]:在1~i物品中,不超过j容量限制时的最大价值.
---------- 普通方法
不选:dp[i][j] = dp[i - 1][j];
选:j >= w[i] -> dp[i][j] = dp[i - 1][j - w[i]] + v[i];
求解顺序: i:1~n;j:0~W;
答案:dp[n][W];
---------- 滚动数组优化
奇数[][][][][][]
偶数[][][][][][]
2行即可
---------- 一维数组优化
[][required][][required][][]
[][]        [][dp]      [][]
i:1~n
	j:w~w[i]
		dp[j] = max(dp[j],dp[j - w[i]] + v[i]);
不推荐!!!可能出错/不适用!!!

完全背包:
有n类体积为t[i],价值为p[i]的物品,问在不超过容量限制下,能取得的物品最大价值.
dp[i][j]:在1~i物品中,不超过j容量限制时的最大价值.
---------- 普通方法
#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int p[10005],t[10005],dp[2][10005];

int main() {
	int m,n;
	cin >> m >> n;
	for (int i = 1;i <= n;i++) cin >> p[i] >> t[i];
	dp[0][0] = 0;
	for (int i = 1;i <= n;i++)
		for (int j = 1;j <= m;j++)
		{
			dp[i][j] = dp[i - 1][j];
			if (j >= t[i]) dp[i][j] = max(dp[i][j],dp[i][j - t[i]] + p[i]); 
		}
	cout << dp[n][m] << endl;
	return 0;
}
不推荐!!!很有可能超时!!!
---------- 滚动数组优化
#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int p[10005],t[10005],dp[2][10005];

int main() {
	int m,n;
	cin >> m >> n;
	for (int i = 1;i <= n;i++) cin >> p[i] >> t[i];
	dp[0][0] = 0;
	for (int i = 1;i <= n;i++)
		for (int j = 1;j <= m;j++)
		{
			dp[i & 1][j] = dp[i - 1 & 1][j];
			if (j >= t[i]) dp[i & 1][j] = max(dp[i & 1][j],dp[i & 1][j - t[i]] + p[i]); 
		}
	cout << dp[n & 1][m] << endl;
    return 0;
}

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=2e3+10,mod=1e9+7;
string s[N];
int n,k;
int num,a[N];
ll f[N],C[N][N];

ll cal(int n,int m)
{
   if(C[n][m]>0)return C[n][m];
   if(n==0||m==0||n==m)return C[n][m]=1;
   return C[n][m]=(cal(n-1,m-1)+cal(n-1,m))%mod;
}

int main()
{
    ios::sync_with_stdio(false); cin.tie(0);
    cin>>n>>k;
    for(int i=1;i<=n;i++)
    {
        cin>>s[i];
        sort(s[i].begin(),s[i].end());
    }
    sort(s+1,s+n+1);
    s[0]='\0';
    for(int i=1;i<=n;i++)
        if(s[i]!=s[i-1])a[++num]=1;
        else a[num]++;

    f[0]=1;
    for(int i=1;i<=num;i++)
    {
        for(int j=k;j>=0;j--)
        {
            for(int x=1;x<=a[i] && x*(x-1)/2<=j;x++)
                f[j]=(f[j]+f[j-x*(x-1)/2]*cal(a[i],x)%mod)%mod;
        }
    }
    cout<<f[k];
    
    return 0;
}
```

package Lanqiao1;

import java.io.;
import java.util.;

java版本

public class Dijkstra求最短路2 {
// 最大的点数量
static int N = 1000010;
// 点的数量和边的数量
static int n,m ;
// 邻接矩阵
static ArrayList []G = new ArrayList[N];
// 存储起点到 某个点的最短距离
static int [] dis = new int[N];
//不可达
static int INF =Integer.MAX_VALUE;

static class Pair{
	int first,second;

	public Pair(int first, int second) {
	
		this.first = first;
		this.second = second;
	} 
	
}

// dijkstra
static void dijkstra(int s) {
// 按照边的权重从小到大排列
PriorityQueue q= new PriorityQueue(Comparator.comparingInt(a->a.second));
// 初始话所有系欸但为不可到达
Arrays.fill(dis, INF);
//起点距离是0
dis[s] =0;
// 将起点添加到队列中
q.add(new Pair(s, 0));
// 当队列非空的时候
while(!q.isEmpty()) {
int u = q.peek().first;
int currentDis =q.peek().second;

		q.poll();

// 直到找到最短路径
if(currentDis!=dis[u] )continue;

		for(Pair edge:G[u]) {
			int v =edge.first;
			int w =edge.second;

// 更新最短距离
if(dis[u]+w<dis[v]) {
dis[v] =dis[u]+ w;
// 最短距离 更新到优先队列中
q.add(new Pair(v, dis[v]));
}
}

	}
	
}

public static void main(String[] args) {
	n =in.nextInt();m =in.nextInt();

// 要先初始化邻接矩阵
for(int i=1; i<=n ; i++) {
G[i] =new ArrayList();
}
// 输入权重
while(m-->0) {
int u =in.nextInt();
int v = in.nextInt();
int w = in.nextInt();
//跟新u-V的距离
G[u].add(new Pair(v, w));

	}

// 起点1
dijkstra(1);
// 如果不存在最短路径 就输出-1 ；
out.print(dis[n]!=INF?dis[n]:-1);

	// TODO Auto-generated method stub
	out.close();

}

static FastReader in = new FastReader();
static PrintWriter out =new PrintWriter(System.out);

	BufferedReader br;
	StringTokenizer st;
	
	public FastReader() {
		br =new BufferedReader(new InputStreamReader(System.in));
		// TODO Auto-generated constructor stub
	}

		String str = "";
		while (st==null||!st.hasMoreElements()) {

			}
			st = new StringTokenizer(str);
		}
		return st.nextToken();
		
	}
	int nextInt() {
		return Integer.parseInt(next());
	}
	
}

}

Dijkstra证明未选节点中路径最短的那个节点u所取的值就是从源s到它的全图最短路径证明上，考虑：如果这个u还没得到它的最短路径，那一定是先到其余未选节点（假设是v）中有一条路到它还能更短，（因为已选节点的出边已经松弛过了），但是因为没有负权路，s到v的路径就比到u长，从v到u又是非负值，不可能比从源S到u的目前值更小。
johnson证明边无负权说的是三角形不等式，实际上是从定义出发,h(v)是bellman求出来的最小值，所以从其他点到v一定大于等于h(v)
